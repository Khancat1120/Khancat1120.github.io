# 无root权限安装软件

# 源代码安装

  1. 首先下载想安装的软件的源码
wget 你想下载的源码压缩包的链接
tar -xzf *.tar.gz
tar -xvf *.tar
  1. 进入bin文件夹，执行以下命令
./configure --prefix=/home/yourname/software/bin
make
make install
  1. 配置环境变量
vim ~/.bashrc
export PATH=$PATH:/home/yourname/software/bin
source ~/.bashrc

# deb安装

  1. 首先下载想安装的软件的deb包
wget 你想下载的源码压缩包的链接
  1. 安装软件包
dpkg -x *.deb /home/yourname/software
